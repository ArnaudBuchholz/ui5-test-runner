```ts
import type { Writable } from './typeUtilities.js';

export type UISelectableItem<KEY_TYPE extends string | number> = {
  label: string;
  key: KEY_TYPE;
};

/** State changes are processed before executing action */
export type UIEvent<State, Actions> = Partial<Writable<State>> & {
  action?: Actions;
};

export type IUserInterfaceController<Settings extends object, State extends object, Actions extends string> = {
  /**
   * Connect the UI to the controller.
   * Everytime the control requires an update in the UI, the update callback is called with state values that changed
   *
   * @returns initial state and settings for static values (drop downs)
   */
  connect(update: (event: Partial<State>) => void): { initialState: State; settings: Settings };

  /**
   * Transmits user interaction to the controller by sending values that must change in the state and/or an action that was triggered
   */
  interaction(event: UIEvent<State, Actions>): void;
};
```

- Inline error display → errorMessage (read-only, provided by controller updates)
- Results list / rows → logs (readonly; InternalLogAttributes[])
- Metrics bar (chunks, sizes, counts) → metrics (LogMetrics)
- Result count / displayed rows → (derived from logs / metrics.logCount) — primary source: logs and metrics
- Selected-row details view → reflected in logs array; controller may expose selected row via a separate state field if implemented (not present in State here)

# Question

My assumption was that if the UI sends an interaction, there is no update being triggered unless really necessary. How should I change the definition to make it obvious ?

# LLM

Make it explicit in the API that interaction may or may not produce immediate UI updates by changing the signature and docs so callers get an explicit response when an update is required.

Two simple, compatible options (pick one):

Option A — interaction returns the changed state fragment (sync)
- Benefit: callers immediately know what changed (or undefined if nothing).
Type change:
```ts
interaction(event: UIEvent<State, Actions>): Partial<State> | undefined;
```
Docs: "Returns the state fragment that changed and should be applied to the UI, or undefined if no UI update is required."

Option B — interaction returns a Promise (async operations)
- Benefit: supports async validation/side-effects before deciding to update.
Type change:
```ts
interaction(event: UIEvent<State, Actions>): Promise<Partial<State> | undefined>;
```
Docs: "Resolves to the state fragment to apply to the UI, or undefined if no UI update is required."

Optional additions (recommended)
- Add a typed enum or union for explicit outcomes instead of undefined, e.g.:
```ts
type InteractionResult<State> = { updated: Partial<State> } | { noUpdate: true };
interaction(event: UIEvent<State, Actions>): InteractionResult<State> | Promise<InteractionResult<State>>;
```
- Improve UIEvent with an optional metadata field to indicate intent (e.g., optimistic UI): `meta?: { expectUpdate?: boolean }`.

Minimal example usage (Option A):
```ts
const changed = controller.interaction({ autorefresh: true });
if (changed) update(changed); // UI applies only when necessary
```

Pick A for simplicity and perf; pick B if controllers perform async validation/network work before deciding to emit updates.

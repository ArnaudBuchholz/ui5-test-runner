```ts
import type { Writable } from './typeUtilities.js';

export type UISelectableItem<KEY_TYPE extends string | number> = {
  label: string;
  key: KEY_TYPE;
};

/** State changes are processed before executing action */
export type UIEvent<State, Actions> = Partial<Writable<State>> & {
  action?: Actions;
};

export type IUserInterfaceController<Settings extends object, State extends object, Actions extends string> = {
  /**
   * Connect the UI to the controller.
   * Everytime the control requires an update in the UI, the update callback is called with state values that changed
   *
   * @returns initial state and settings for static values (drop downs)
   */
  connect(update: (event: Partial<State>) => void): { initialState: State; settings: Settings };

  /**
   * Transmits user interaction to the controller by sending values that must change in the state and/or an action that was triggered.
   *
   * NOTE: The UI should optimistically assume the controller accepts the change. The UI MUST send only the changed fields
   * in the UIEvent payload. The controller only calls the update(...) callback when it needs to correct or reject a change —
   * i.e., to restore previous values or to surface validation errors. If the controller accepts the change, it does not need
   * to call update(...) to confirm.
   */
  interaction(event: UIEvent<State, Actions>): void;
};
```

## Example interaction sequence: enable auto-refresh and set interval to 60s

Assumptions: autorefreshInterval keys are numeric seconds (e.g., 10, 60). UI calls are optimistic; controller only calls the update callback to correct/reject changes.

1) UI connects to controller (on mount)
```ts
// UI calls connect(...)
const { initialState, settings } = controller.connect(updateCallback);
```
Controller returns initialState (example)
```ts
initialState = {
  timerangeType: 'relative',
  relativeTimerange: 15,
  absoluteTimerangeFrom: 0,
  absoluteTimerangeTo: 0,
  autorefresh: false,
  autorefreshInterval: 10,
  filter: '',
  errorMessage: '',
  logs: [],
  metrics: { inputSize:0, chunksCount:0, outputSize:0, minTimestamp:0, maxTimestamp:0, logCount:0, reading:false }
};
```

2) User clicks the "Auto refresh" checkbox (turn ON)
- UI sends optimistic interaction event (only changed fields):
```ts
controller.interaction({ autorefresh: true });
```
- Expected controller behavior:
  - If accepted: controller does nothing (UI remains optimistic).
  - If rejected/validated: controller calls updateCallback with a partial state to correct or set error, for example:
```ts
updateCallback({ autorefresh: false, errorMessage: 'Auto-refresh not allowed for this range' });
```

3) User opens Interval dropdown and selects "60s"
- UI sends event for changed field:
```ts
controller.interaction({ autorefreshInterval: 60 });
```
- Expected controller behavior:
  - If accepted: no update callback is required.
  - If controller needs to validate (e.g., min interval 30s) it may respond:
```ts
updateCallback({ autorefreshInterval: 30, errorMessage: 'Minimum interval is 30s' });
```

4) Controller starts background refresh loop (server-side) if autorefresh enabled
- Controller may update `metrics.reading` or `metrics` and `logs` as new data arrives by calling:
```ts
updateCallback({ metrics: { ...updatedMetrics }, logs: [...newLogs], errorMessage: '' });
```

Notes:
- Each interaction event must only include the changed fields per UIEvent contract.
- Actions (like pressing "Refresh Now") are sent as `{ action: 'refresh_now' }`.
